% Custom header/footer branding for DocCraft PDF
% This file is included in the LaTeX preamble

% Page headers and footers
\usepackage{fancyhdr}
\usepackage{graphicx}
\usepackage{lastpage}

% Configure fancy headers
\pagestyle{fancy}
\fancyhf{} % Clear all header and footer fields

% Header configuration
\fancyhead[L]{\small\textit{\thetitle}}
\fancyhead[R]{\includegraphics[height=0.8cm]{assets/logo.png}}

% Footer configuration
\fancyfoot[L]{\small Generated by DocCraft PDF}
\fancyfoot[C]{\small\thedate}
\fancyfoot[R]{\small Page \thepage\ of \pageref{LastPage}}

% Adjust header rule
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.4pt}

% First page style (plain)
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\small Page \thepage\ of \pageref{LastPage}}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0.4pt}
}

% Adjust spacing
\setlength{\headheight}{15pt}

% Colors for links and sections
\usepackage{xcolor}
\definecolor{brandblue}{RGB}{0,102,204}
\definecolor{brandgray}{RGB}{64,64,64}

% Section formatting
\usepackage{titlesec}
\titleformat{\section}
  {\color{brandblue}\Large\bfseries}
  {\thesection}{1em}{}
\titleformat{\subsection}
  {\color{brandblue}\large\bfseries}
  {\thesubsection}{1em}{}
